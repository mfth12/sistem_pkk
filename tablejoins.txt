SELECT 
    berita.*, 
    pokja.nama_pokja, 
    users.nama
FROM
    berita
LEFT JOIN pokja
    ON pokja.id_pokja = berita.id_pokja;
LEFT JOIN users
    ON users.id_user = berita.id_user;
ORDER BY 
    id_berita, DESC
